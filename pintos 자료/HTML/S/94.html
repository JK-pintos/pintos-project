  60 /* Formats the printf() format specification FORMAT with

  62    HANDLE. */
  63 int
  64 vhprintf (int handle, const char *format, va_list args) 
  65 {
  66   struct vhprintf_aux aux;
  67   aux.p = aux.buf;
  68   aux.char_cnt = 0;

  70   __vprintf (format, args, add_char, &aux);
